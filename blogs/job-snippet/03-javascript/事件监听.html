# 事件监听

# scroll监听改变背景色

window.addEventListener('scroll', () => {
  const scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
  dom.style.backgroundColor = scrollTop > 50 ? rgba(255, 0, 0, 1) : 'transparent';
});

# pagehide监听暂停视频

window.addEventListener('visibilitychange', () => {
  const { vedioRef } = this.$refs;
  if (vedioRef && !vedioRef.paused) {
    vedioRef.pause();
  }
})
