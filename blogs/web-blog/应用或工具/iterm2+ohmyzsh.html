# 主题存储路径

# 访达 -> 前往文件夹
/Users/yourmacname/.oh-my-zsh/themes/

# 或者命令行
cd ~/.oh-my-zsh/themes && ls

# 修改主题

# 打开配置文件
vim ~/.zshrc

# 修改配置项, ys 既主题名
ZSH_THEME="ys"

# 使配置生效
source ~/.zshrc

# 修改某主题配置

cd ~/.oh-my-zsh/themes
vim ./colud.zsh-theme

# 常见修改项
echo "删除echo中的无用空格"
# 前缀显示的用户名
prompt_segment black default ""
# 电脑用户名
%u
# 文件路径
%c
# 文件完整路径
%PWD

# 安装插件

# 打开配置文件
vim ~/.zshrc

# 修改配置项，配置插件名
plugins=(
  git
  zsh-autosuggestions
)

# 使配置生效
source ~/.zshrc

# 配置别名

# 打开配置文件
vim ~/.zshrc

# 修改配置项
alias ll='ls -l'
alias la='ls -a'
alias ga='git add .'
alias gc='git cz'
alias gl='git pull'
alias gh='git push'
# 只切换分支
alias gco='git checkout'
# 创建新分支并切换
alias gcb='git checkout -b'
# 合并代码
alias gm='git merge'

# 使配置生效
source ~/.zshrc
