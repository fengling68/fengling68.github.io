# flex

仅个人学习总结之作，可能存在问题，使用过程中会不断优化。

/**
 * 时间：2019-5-7
 * 参考文档：https://segmentfault.com/a/1190000003978624
 */
/**
 * 作用：给父盒子添加flex属性
 * 注意：如果子元素是行内元素，多数使用 display: block 或 display: inline-block 把行内子元素变成块元素。
 */
@mixin mix-flex {
  display: -webkit-flex;
  display: -moz-flex;
  display: flex;
  display: -webkit-box;
  display: -moz-box;
  display: -ms-flexbox;
}
@mixin mix-flex-son {
  -webkit-flex: 1;
  -ms-flex: 1;
  flex: 1;
  -webkit-box-flex: 1;
  -moz-box-flex: 1;
}
/**
 * 作用：flex-direction，子元素在主轴的排列方向
 * 传值：row（左->右）| row-reverse（右->左）| column（上->下）| column-reverse（下->上）
 */
@mixin mix-flex-direction($d: row) {
  @if $d == 'row' {
    -webkit-box-direction: normal;
    -webkit-box-orient: horizontal;
  } @else if $d == 'row-reverse' {
    -webkit-box-pack: end;
    -webkit-box-direction: reverse;
    -webkit-box-orient: horizontal;
  } @else if $d == 'column' {
    -webkit-box-direction: normal;
    -webkit-box-orient: vertical;
  } @else if $d == 'column-reverse' {
    -webkit-box-pack: end;
    -webkit-box-direction: reverse;
    -webkit-box-orient: vertical;
  } 
  -webkit-flex-direction: $d;
  -moz-flex-direction: $d;
  -ms-flex-direction: $d;
  -o-flex-direction: $d;
  flex-direction: $d;
}
/**
 * 作用：justify-content，子元素在主轴对齐方式；box-pack(旧版4个值): start | end | center | justify;
 * 传值(5个值)：flex-start(默认) | flex-end | center | space-between | space-around
 */
@mixin mix-justify-content($p: flex-start) {
  @if $p == 'flex-start' {
    -webkit-box-pack: start;
  } @else if $p == 'flex-end'{
    -webkit-box-pack: end;
  } @else if $p == 'center'{
        -webkit-box-pack: $p;
  } @else if $p == 'space-between'{
    -webkit-box-pack: justify;
  } @else if $p == 'space-around'{
    // box-pack 不可用
  }
  -webkit-justify-content: $p;
  -moz-justify-content: $p;
  -ms-justify-content: $p;
  -o-justify-content: $p;
  justify-content: $p;
}
/**
 * 作用：align-items，子元素在交叉轴轴对齐方式；box-align(旧版4个值): start | end | center | baseline | stretch;
 * 传值(5个值)： flex-start | flex-end | center | baseline | stretch;
 */
@mixin mix-align-items($p: flex-start) {
  @if $p == 'flex-start' {
    -webkit-box-align: start;
  } @else if $p == 'flex-end'{
    -webkit-box-align: end;
  } @else {
    -webkit-box-align: $p;
  }
  -webkit-align-items: $p;
  -moz-align-items: $p;
  -ms-align-items: $p;
  -o-align-items: $p;
  align-items: $p;
}
/**
 * 作用：flex-wrap，元素是否换行
 * 传值：nowrap | wrap | wrap-reverse
 */
@mixin mix-flex-wrap($wrap: nowrap) {
  -webkit-flex-wrap: $wrap;
  -moz-flex-wrap: $wrap;
  -ms-flex-wrap: $wrap;
  -o-flex-wrap: $wrap;
  flex-wrap: $wrap;
}
/**
 * 作用：flex-grow，是否放大
 * 传值：0（不放大）| 1（放大）
 */
@mixin mix-flex-grow($value: 0) {
  @if $value == 0 {
    // 不伸缩
    -webkit-box-flex: 0.0;
  } @else if $value == 1 {
    -webkit-box-flex: 1.0;
  } @else {
    -webkit-box-flex: $value;
  }
  -webkit-flex-grow: $value;
  -moz-flex-grow: $value;
  flex-grow: $value;
}
/**
 * 作用：flex-shrink，是否缩小
 * 传值：0（不缩小）| 1（缩小）
 */
 @mixin mix-flex-shrink($value: 1) {
  @if $value == 0 {
    // 不伸缩
    -webkit-box-flex: 0.0;
  } @else if $value == 1 {
    -webkit-box-flex: 1.0;
  } @else {
    -webkit-box-flex: $value;
  }
  -webkit-flex-shrink: 1;
  -moz-flex-shrink: 1;
  flex-shrink: 1;
}
/**
 * 作用：order，子元素排序
 */
 @mixin mix-order($o) {
  -webkit-box-ordinal-group: $o;
  -webkit-order: $o;
  -moz-order: $o;
  order: $o;
}
