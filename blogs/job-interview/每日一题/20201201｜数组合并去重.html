# 数组合并去重

function combo (arr1, arr2) {
  return Array.from(new Set([...arr1, ...arr2]))
}
console.log(combo([1, 2, 3], [3, 4, 5]));

function combo (arr1, arr2) {
  return [...new Set([...arr1, ...arr2])];
}
console.log(combo([1, 2, 3], [3, 4, 5]));
