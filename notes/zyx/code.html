# 代码片段

(function (hasTitle, psdWidth) {
  psdWidth = psdWidth || 640;
  var landscape = ''; // 横屏
  var portrait = ''; // 竖屏
  var timer = null;
  var zoomStyle = document.getElementById('zoom-style');
  if (!zoomStyle) {
    zoomStyle = document.createElement('style');
    zoomStyle.id = 'zoom-style';
    zoomStyle.type = 'text/css';
    document.documentElement.appendChild(zoomStyle);
  }
  _resize();
  window.addEventListener('orientationchange', resize, false);

  function resize() {
    clearTimeout(timer);
    timer = setTimeout(_resize, 500);
  }

  function _resize() {
    var screenWidth = document.documentElement.clientWidth;
    var screenHeight = window.innerWidth !== screenWidth
      ? window.innerHeight * (screenWidth / window.innerWidth)
      : window.innerHeight;
    var fontSize = screenWidth/ (psdWidth / 2) * 50;

    if (!screenWidth) { // QQ浏览器有机率为0
      setTimeout(_resize, 100);
      return;
    }

    if (screenWidth > screenHeight) { // 横屏
      landscape = `
        @media all and (orientation: landscape) {
          html {
            font-size: ${fontSize}px!important;
          }
        }
      `;
    } else { // 竖屏
      var scaleSize = Math.min(1, screenHeight / screenWidth * (hasTitle ? 750 / 1207 : 750 / 1334));
      portrait = `
        html { font-size: ${fontSize}px!important;}
        ${
          scaleSize < 1
            ? `.auto_scale {-webkit-transform: scale(${fontSize}); transform: scale(${fontSize});}.auto_scale_center {-webkit-transform: translate(-50%, -50%) scale(${fontSize});transform: translate(-50%, -50%) scale(${fontSize});}`
            : `.auto_scale_center { -webkit-transform: translate(-50%, -50%);transform: translate(-50%, -50%);}`
        }
      `;
    }

    zoomStyle.innerHTML = `${landscape}${portrait}`;
    if (landscape && portrait) {
      window.removeEventListener('orientationchange', resize);
    }
  }
})(true, 750)

# 滚动改变透明tittlebar背景

let scrollTimer = null;
window.addEventListener('scroll', () => {
  if (scrollTimer) return;
  scrollTimer = setTimeout(() => {
    const current = document.documentElement.scrollTop || document.body.scrollTop || 0;
    if (current >= 30) {
      this.titleData.backgroundColor = '#fdd0a4';
    } else {
      this.titleData.backgroundColor = 'transparent';
    }
  }, 50);
})
