function clone (data) {
  const t = Object.prototype.toString.call(data);
  const type = t.replace('[object ', '').replace(']', '').toLowerCase();

  let copyData;

  if (type === 'object' || type === 'array') {
    copyData = type === 'object' ? {} : [];
    for(let key in data){
      copyData[key] = clone(data[key])
    }
  } else {
    copyData = data;
  }

  return copyData;
}
