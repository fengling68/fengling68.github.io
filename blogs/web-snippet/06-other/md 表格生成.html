# md 表格生成

本示例并不具备通用性，仅在一次场景中使用到，此处记录下，以备查阅。

var data = [
  'a', 'b', 'c', 'd'
];
var rowLen = 2;
var str = '';
for(var i = 1; i <= data.length; i++){
  if (i % rowLen === 1) {
    str += '| `'+data[i-1]+'` | '+data[i-1]+' |';
  } else if (i % rowLen === 0) {
    str += ' `'+data[i-1]+'` | '+data[i-1]+' | \n' ;
  } else {
    str += ' `'+data[i-1]+'` | '+data[i-1]+' |';
  }
}
console.log(str);

// 结果
| `a` | a | `b` | b | 
| `c` | c | `d` | d |
