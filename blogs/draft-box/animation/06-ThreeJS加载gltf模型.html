# ThreeJS读取gltf文件渲染3D模型

# 实现功能

  • vue组件
  • 加载 gltf 模型
  • 一键换肤
  • 截图

# 代码

<template>
  <div class="wrap">
    <canvas id="stage" class="stage"></canvas>

    <div class="btn" @click="changeSkin">一键换肤</div>
    <div class="btn" @click="doCapture">截图</div>
    <div class="capture-img">
      <img :src="carCaptureImg" />
    </div>
  </div>
</template>

<script>
import {
  Scene,
  WebGLRenderer,
  PerspectiveCamera,
  AmbientLight,
  DirectionalLight,
  AnimationMixer,
  Color,
  Clock,
  CubeTextureLoader,
  Mesh,
  MeshPhongMaterial,
} from 'three';
import { TextGeometry } from 'three/examples/jsm/geometries/TextGeometry';
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls';
import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader';
import { FontLoader } from 'three/examples/jsm/loaders/FontLoader';

// three
let scene = null;
let camera = null;
let renderer = null;
let controls = null;
const light = {};
const ambientLight = {};

// model
let gltfLoader = null;
let gltfModel = null;
let gltfMaterial = null;

// animation
let mixer = null;
const clock = new Clock();

export default {
  name: 'demo3D',
  data() {
    return {
      carCaptureImg: '',
    };
  },
  async mounted() {
    // 创建舞台
    const initRes = await this.initScene().catch(() => false);
    if (!initRes) return;

    // 处理模型
    const gltfRes = await this.handleGltfLoader().catch(() => false);
    if (!gltfRes) return;

    // 周围场景
    this.handleSceneBg();

    // 执行渲染
    this.render();
  },
  methods: {
    // 创建舞台和渲染器
    initScene() {
      return new Promise((resolve, reject) => {
        // canvas 
        const canvasDom = document.querySelector('#stage');
        if (!canvasDom) {
          reject(new Error(false));
          return;
        }
        const domWidth = canvasDom.offsetWidth;
        const domHeight = canvasDom.offsetHeight;

        // 场景
        scene = new Scene();
        scene.background = new Color(0x00ff80);

        // 渲染器
        renderer = new WebGLRenderer({
          canvas: canvasDom,
          antialias: true, // 开启反锯齿
          alpha: true, // 设置背景透明色
          preserveDrawingBuffer: true, // 保留图片缓冲区方便截图
        });
        renderer.setSize(domWidth, domHeight);
        renderer.setPixelRatio(window.devicePixelRatio);

        // 相机
        camera = new PerspectiveCamera(40, domWidth / domHeight, 1, 1000);
        camera.position.set(0, 0, 0);
        camera.lookAt(scene.position);

        // 控制器
        controls = new OrbitControls(camera, renderer.domElement);
        controls.maxDistance = 200; // 最远距离
        controls.minDistance = -200; // 最近距离
        // controls.addEventListener('change', this.render);

        // 环境光
        ambientLight.light1 = new AmbientLight(0xffffff);
        scene.add(ambientLight.light1);

        // 灯光
        const lightPosArr = [
          { color: 0xffffff, pos: [0, 100, 0] },
          { color: 0xffffff, pos: [100, 0, 100] },
          { color: 0xffffff, pos: [0, -100, 0] },
        ];
        lightPosArr.forEach((item, index) => {
          const { color, pos } = item;
          const [x, y, z] = pos;
          let curLight = light[`light${index}`];
          curLight = new DirectionalLight(color, 0.8);
          curLight.position.set(x, y, z);
          scene.add(curLight);
        });

        resolve(true);
      });
    },

    // 处理模型
    handleGltfLoader() {
      return new Promise((resolve, reject) => {
        gltfLoader = new GLTFLoader();
        gltfLoader.load('../dog.gltf', (model) => {
          // 添加模型到场景
          gltfModel = model;
          gltfModel.scene.position.set(0, 0, 0);
          gltfModel.scene.scale.set(10, 10, 10);
          scene.add(gltfModel.scene);

          // 动画处理
          mixer = new AnimationMixer(gltfModel.scene);
          mixer.clipAction(gltfModel.animations[0]).play();

          // 输出返回
          console.log('模型加载成功', model, model.parser.json, model.parser.json.nodes);
          resolve(true);
        }, (xhr) => {
          console.log(`模型加载进度: ${(xhr.loaded / xhr.total) * 100}%`);
        }, (err) => {
          console.log(`模型加载失败: ${err}`);
          reject(new Error());
        });
      });
    },

    // 处理环境
    handleSceneBg() {
      const bgImgs = [
        'https://t7.baidu.com/it/u=3876201379,2104650373&fm=193&f=GIF',
        'https://t7.baidu.com/it/u=3876201379,2104650373&fm=193&f=GIF',
        'https://t7.baidu.com/it/u=3876201379,2104650373&fm=193&f=GIF',
        'https://t7.baidu.com/it/u=3876201379,2104650373&fm=193&f=GIF',
        'https://t7.baidu.com/it/u=3876201379,2104650373&fm=193&f=GIF',
        'https://t7.baidu.com/it/u=3876201379,2104650373&fm=193&f=GIF',
      ];
      new CubeTextureLoader().load(bgImgs, (cubeTexture) => {
        scene.background = cubeTexture;
      });
    },

    // 渲染函数
    render() {
      // 无动画处理
      // renderer.render(scene, camera);

      // 有动画处理
      requestAnimationFrame(this.render);
      const delta = clock.getDalta();
      if (mixer != null) {
        mixer.update(delta);
      }
      mixer.update(clock.getDalta());
      renderer.render(scene, camera);
    },

    // 写文字
    drawText(text = '123456') {
      new FontLoader().load('../gentilis_blod.typeface.json', (font) => {
        const textGeometry = new TextGeometry(text, {
          font,
          size: 1,
          height: 1,
        });
        textGeometry.center();
        const textMesh = new Mesh(textGeometry, [
          new MeshPhongMaterial({ color: 0xff0000, flatShading: true })
        ]);
        textMesh.position.y = 1;
        textMesh.rotation.y = Math.PI * 1;

        // 获取gltf模型材料 & 添加文字
        const { scene: gltfScene } = gltfModel;
        const mesh = gltfScene.getObjectByName('name');
        mesh.add(textMesh);
      });
    },

    // 一键换肤
    changeSkin() {
      const r = this.randomNum(0, 255);
      const g = this.randomNum(0, 255);
      const b = this.randomNum(0, 255);

      // 获取车壳 & 换肤
      if (!gltfMaterial) {
        const { scene: gltfScene } = gltfModel;
        const mesh = gltfScene.getObjectByName('name');
        gltfMaterial = mesh.material;
      }
      gltfMaterial.color.setRGB(r, g, b);
    },

    // 随机数
    randomNum(min, max) {
      return parseInt(Math.random() * (max - min + 1) + min, 10);
    },

    // 截图
    doCapture() {
      const canvasDom = document.getElementById('stage');
      if (!canvasDom) return;
      this.carCaptureImg = canvasDom.toDataURL('image/jpeg');
    },
  },
}
</script>

<style>

  height: 50vh;
}
