# 固定格式

// ==UserScript==
// @key value
// ==/UserScript==

# 语法列表

@key 示例值 说明
@name // @name 广告拦截脚本 名称
@version // @version 0.1 版本
@description // @description 一个广告拦截脚本，可拦截...... 说明
@namespace // @namespace http://tampermonkey.net/ 名称空间（可以是任何唯一的值）
@icon // @icon http://www.example.org/icon.png 图标（32 x 32 大小最佳）
@match // @match https://*/* 生效规则
@include // @include https://*/* 包含规则
@exclude // @exclude https://www.baidu.com/* 排除规则
@grant // @grant none 控制脚本使用哪些特殊 api
@require // @require http://www.example.com/example.js 引用任意数量外部脚本
@resource // @resource resourceName http://www.example.com/example.png 引用任意数量资源（唯一名 资源地址）
@noframes // @noframes 该命令将限制脚本的执行，只能在顶级文档中运行，不能在嵌套的框架中运行。
@run-at // @run-at document-idle 脚本运行的时机：document-start（文件启动）、document-end（文件结束）、document-idle（文件空闲）

# 新页面打开a链接

// ==UserScript==
// @name         页面a标签新窗口打开
// @namespace    yishiqingan-fenglinggrass
// @version      0.1
// @description  try to take over the world!
// @author       风铃
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const aLinks = document.querySelectorAll('a');
    aLinks.forEach((ele) => {
        if (ele.getAttribute('href')) {
            ele.setAttribute('target', '_blank');
        }
    });
})();
