3、JS部分

/* 1、引入文件 */
<script src="swiper/swiper.min.js" type="text/javascript" charset="utf-8"></script>
<script type="text/javascript">
    
    /* 2、初始化操作 */
    
    //创建轮播图对象
    //参数一：轮播对象的选择器
    //参数二：配置轮播图的选项
    var mySwiper = new Swiper(".swiper-container",{
        //是否循环滚动
        loop:true,
        //自动滚动时间
        autoplay: 2000,
        //指定分页组件,选择器
        pagination : ".swiper-pagination",
        //点击分页,是否切换图片
        paginationClickable: true,
        
        //变小手
        grabCursor: true,
        //第一次加载的时候，加载的slide序号
        initialSlide: 1,
        //改变方向——垂直切换 默认水平
        direction: "vertical"，

         //切换效果
        //slide:滑动  fade:淡入淡出  cube：立方体  coverflow：封面浏览  flip：轻弹
        effect: "slide"

        //其他配置项参考说明文档
        ...
    })
    
</script>

<!--轮播图容器  名字是固定的-->
<div class="swiper-container">            
    <!--轮播的图片-->
    <div class="swiper-wrapper">            
        <div v-for="item in adsList" class="swiper-slide">
            <img v-bind:src="imagePrefix+item.img_url"/>
        </div>           
    </div>            
    <!--底部的控制按钮-->
    <div class="swiper-pagination"></div>
</div>

4、Vue的js部分

created:function(){
    ......

    /* 下载广告数据 url后是广告的地址 */
    this.$http({url:adsUrl}).then(function(response){
        var list = response.data.data;
        this.adsList = list;
        /* Vue有一个虚拟dom机制, 数据改变, 界面自动变化 */
        /* Vue.$nextTick()表示界面更新完成的下一个阶段 */
        Vue.nextTick(function(){
            var mySwiper = new Swiper(".swiper-container",{
                //是否循环滚动
                loop:true,
                //自动滚动时间
                autoplay: 2000,
                //指定分页组件,选择器
                pagination : ".swiper-pagination",
                //点击分页,是否切换图片
                paginationClickable: true,
                //变小手
                grabCursor: true,
                //第一次加载的时候，加载的slide序号
                initialSlide: 1,
                //垂直切换 默认水平
                direction: "vertical"，

                 //切换效果
                //slide:滑动  fade:淡入淡出  cube：立方体  coverflow：封面浏览  flip：轻弹
                effect: "slide"                    
            })
        })
    })
}
