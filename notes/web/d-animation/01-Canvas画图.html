# Canvas和SVG的区别？

  1. 位图和矢量图的区别

  2. 位图色彩更丰富，

  3. 矢量图色彩较单一，经常做图标

     <!-- canvas 点击鼠标右键，出现图片另存为 -->
 <!-- canvas 只能写在标签内，不能放在CSS样式中 -->
 <canvas id="myCanvas" width="500px" height="400px"></canvas>

# 一、画线

//1、获取 canvas元素
var myCanvas = document.querySelector("#myCanvas");

//2、获取上下文：简单理解，把canvas画布变成2D画板。 WebGL：3D画板
var ctx = myCanvas.getContext("2d");

//3、设置样式
ctx.lineWidth = 10;
ctx.strokeStyle = "greenyellow";

//4、移动至某一点
ctx.moveTo(100,100);

//5、创建一个新点 两个操作，移动到这个点，连接前一个点
ctx.lineTo(100,200);
ctx.lineTo(100,350);
ctx.lineTo(300,350);
ctx.lineTo(300,100);

//6、画线
ctx.stroke();

//7、画另一条线
ctx.moveTo(200,300);
ctx.lineTo(200,100);
ctx.lineTo(200,100);
ctx.lineTo(400,50);
ctx.stroke();

# 二、画矩形

//前两部相同
//3、x,y：矩形的起点      width,height:矩形的宽高
ctx.fillStyle = "red";
ctx.fillRect(50,50,100,50);

ctx.rect(200,50,100,50);
ctx.fillStyle = "aqua";
ctx.fill();

ctx.rect(50,200,100,50);
ctx.stroke();

ctx.strokeStyle = "blueviolet";
ctx.strokeRect(200,200,100,50);

# 三、写字

//3、设置样式
ctx.font = "50px 微软雅黑";
ctx.strokeStyle = "gold";

//4、参数一：文字内容    参数二：文字显示的坐标
//爬虫爬取不到canvas的文本内容
ctx.fillText("秦时明月之君临天下",20,100);
ctx.strokeText("秦时明月汉时关",80,200);

# 四、画圆

/*3、参数一二：圆心坐标
 *   参数三    ：半径
 *   参数四五：圆的起始和结束弧度
 *   参数六 是否逆时针   ：默认 false————顺时针
 * */
ctx.arc(300,300,100,0,360,false);
ctx.fillStyle = "chartreuse";
ctx.fill();

ctx.arc(100,100,100,0,360,false);
ctx.stroke();

# 五、画弧线、扇形

/*
 * 角度和弧度的转化 2π = 360(弧度)    π = 180
 * Math.PI / 180 * n
 */
ctx.arc(300,300,80,0,3.14,false);
ctx.fillStyle = "chartreuse";
ctx.fill();

//解决路径闭合问题：方法一：下一个圆的圆心+半径   方法二：关闭路径
ctx.moveTo(180,100);   //可注释后查看效果

ctx.arc(100,100,80,0,3.14,false);
ctx.stroke();

注意：画扇形，需要moveTo(原点，原点);

# 六、线性渐变

//填充渐变颜色
var gradient = ctx.createLinearGradient(0,0,500,400);

//参数一：颜色位置 0~1 之间； 参数二：颜色值
gradient.addColorStop(0,"red");
gradient.addColorStop(0.3,"orange");
gradient.addColorStop(1/2,"green");    //注意：可以是分号
gradient.addColorStop(0.8,"yellow");
gradient.addColorStop(1,"purple");

//设置样式——填充渐变色
ctx.fillStyle = gradient;

//画个矩形等
ctx.fillRect(0,0,500,400);

# 七、经向渐变

//填充渐变颜色
var gradient = ctx.createRadialGradient(200,200,50,200,200,200);

//参数一：颜色位置 0~1 之间；参数二：颜色值
gradient.addColorStop(0,"red");
gradient.addColorStop(0.2,"orange");
gradient.addColorStop(0.4,"yellow");
gradient.addColorStop(0.6,"green");
gradient.addColorStop(0.8,"yellow");
gradient.addColorStop(1,"purple");

//设置样式——填充渐变色
ctx.fillStyle = gradient;

//画个矩形等
ctx.arc(200,200,200,0,360);
ctx.fill();

# 八、放入图片

//可以用canvas做截图
//创建图像位置
var img = new Image();
img.src = "img/354279.jpg";
img.onload = function(){
    //参数1：放入img对象
    //参数2,3：图片的开始位置
    //参数4,5：图片的变形  宽度和高度 
    ctx.drawImage(img,10,10,480,380);
}

# 九、路径闭合问题解决方法二

//用beginPath()和closePath()方法，开始、关闭路径
ctx.beginPath();
ctx.arc(100,100,100,0,360,false);
ctx.fillStyle = "red";
ctx.fill();
ctx.closePath();

ctx.beginPath();
ctx.arc(300,300,100,0,360,false);
ctx.fillStyle = "chartreuse";
ctx.fill();
ctx.closePath();
