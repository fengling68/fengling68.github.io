# upload上传

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0, user-scalable=0, minimum-scale=1.0, maximum-scale=1.0, viewport-fit=cover"/>
  <meta name="format-detection" content="telephone=no" />
  <meta http-equiv="Pragma" content="no-cache" />
  <meta http-equiv="Cache-Control" content="no-cache" />
  <meta http-equiv="Expires" content="0" />
  <script src="https://testing.amap.com/js/vconsole.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/zepto@1.2.0/dist/zepto.min.js"></script>
  <style>
    body{
      position: absolute;
      left: 0;
      top: 0;
      margin: 0;
      padding: 10px;
      width: 100%;
      box-sizing: border-box;
      overflow: hidden;
    }
    .title {
      line-height: 60px;
      font-size: 26px;
      font-weight: bold;
    }
    .file {
      width: 200px;
      background: rgb(157, 247, 227);
      border: none;
      border-radius: 5px;
      padding: 10px;
    }
    .btn {
      padding: 10px;
      background: rgb(157, 247, 227);
      border: none;
      border-radius: 5px;
      margin: 20px 0 30px;
    }
    .info {
      width: 100%;
      border: 1px solid ghostwhite;
      padding: 5px;
    }
    video {
      position: absolute;
      left: 100%;
      top: 0;
      width: 100%;
      height: 200px;
      opacity: 0;
      background: pink;
      /* display: none; */
    }
  </style>
</head>
<body>
  <!-- 图片上传 -->
  <div class="title">点击选择图片文件</div>
  <form class="form">
    <input type="file" accept="image/jpeg, image/*" class="file pic-file" multiple/>
  </form>
  <button class="btn audio-btn">点击上传图片文件</button>
  <div class="info" id="picInfo"></div>

  <!-- 视频上传 -->
  <div class="title">点击选择视频文件</div>
  <form class="form">
    <input type="file" accept="video/mp4,video/x-m4v,video/*" class="file video-file" multiple/>
  </form>
  <button class="btn video-btn">点击上传视频文件</button>
  <div class="info" id="videoInfo"></div>

  <script>
    const picFileEle = $(".pic-file");
    const videoFileEle = $(".video-file");
    const picBtnEle = $(".audio-btn");
    const videoBtnEle = $(".video-btn");
    let uploadPic = null;
    let uploadVideo = null;

    // 图片选择
    picFileEle.on("change", function(e) {
      const picFiles = e.target.files;
      uploadPic = picFiles;
      console.log('uploadPic', uploadPic);
    });

    // 点击上传图片
    picBtnEle.on("click", function() {
      if (!uploadPic || uploadPic.length === 0) {
        alert("请选择图片文件");
        return false;
      }

      document.querySelector('#picInfo').innerHTML = ''; 
      const formData = new FormData();
      for (let i = 0, len = uploadPic.length; i < len; i++) {
        const file = uploadPic[i];
        formData.append(file.name, file);
        const info = `【${file.name}】${Math.ceil(file.size / 1024)}kb<br/>`;
        document.querySelector('#picInfo').innerHTML += info; 
      }
      console.log(formData);

      // 接口上传文件
      $.ajax({
        type: "POST",
        url: location.href.indexOf('amap.com') !== -1 ? '线上接口' : 'http://127.0.0.1:7001/upload',
        data: formData,
        processData: false,
        contentType: false,
        processData: false,
        success: function(data) {
          data.forEach(filename => {
            document.querySelector('#picInfo').innerHTML += `资源前缀地址/${filename}<br/>`;
          });
          picFileEle.files = [];
          picFileEle.value = "";
          uploadPic = [];
        },
        error: function(xhr, type) {
          alert("Ajax error!");
        }
      });
    });


    // 视频选择
    videoFileEle.on("change", function(e) {
      const videoFile = e.target.files;
      uploadVideo = videoFile;
      console.log('uploadVideo', uploadVideo);
    });

    // 点击上传视频
    videoBtnEle.on("click", function() {
      if (!uploadVideo || uploadVideo.length === 0) {
        alert("请选择视频文件");
        return false;
      }

      document.querySelector('#videoInfo').innerHTML = ''; 
      const formData = new FormData();
      let loadLength = 0;
      for (let i = 0, len = uploadVideo.length; i < len; i++) {
        const file = uploadVideo[i];
        formData.append(file.name, file);
        // 创建audio标签，获取视频的时长
        var videoSrc = URL.createObjectURL(file);
        let createVideo = document.createElement("video");
        createVideo.setAttribute("src", videoSrc);
        createVideo.setAttribute("preload", "meta");
        document.body.appendChild(createVideo);
        // 所有视频时长获取成功后在上传文件
        createVideo.onloadedmetadata = function () {
          loadLength++;
          const info = `【${file.name}】${createVideo.duration}s  ${Math.ceil(file.size / 1024)}kb<br/>`;
          document.querySelector('#videoInfo').innerHTML += info;
          setTimeout(() => { if (createVideo) createVideo.remove(); }, 2000);
          if (loadLength === uploadVideo.length) {
            console.log(formData);
            $.ajax({
              type: "POST",
              url: location.href.indexOf('amap.com') !== -1 ? '线上接口' : 'http://127.0.0.1:7001/upload',
              data: formData,
              processData: false,
              contentType: false,
              processData: false,
              success: function(data) {
                data.forEach(filename => {
                  document.querySelector('#videoInfo').innerHTML += `资源前缀地址/${filename}<br/>`;
                });
                videoFileEle.files = [];
                videoFileEle.value = "";
                uploadVideo = [];
              },
              error: function(xhr, type) {
                alert("Ajax error!");
              }
            });
          }
        }
        // 自动播放预加载验证
        if (location.href.indexOf('preload=true') !== -1) {
          if (createVideo) createVideo.play();
          setTimeout(() => {
            if (createVideo) createVideo.pause();
          }, 2000);
        }
        // 上传逻辑end
      }
    });
  </script>
</body>
</html>
