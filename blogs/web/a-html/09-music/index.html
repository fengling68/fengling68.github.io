# audio有进度条版

// html部分
<template>
<div class="listen-voice">
  <div class="voice-name">{{topVoiceData.voice_name}}</div>
  <div class="voice-progress" @click="clickPrograssBar($event)">
    <div class="progress-ing">
      <div class="progress-point" @touchmove.self="dragProgressPoint($event)"></div>
    </div>
  </div>
  <div class="voice-duration">{{musicCurTime}}/{{musicDuration}}</div>
  <div class="voice-btn play"  @click="playListAudio('addlog')"  v-if="!isMusicPlaying"></div>
  <div class="voice-btn pause" @click="pauseListAudio('addlog')" v-else></div>
</div>
</template>

<script>
export default {
  // js部分
  data() {
    return {
      isMusicPlaying: false,
      musicCurTime: '00:00',
      musicDuration: '00:00',
      musicLoadTimer: null,
    };
  },
  mounted() {
    this.setMusicDuration();
    window.onpagehide = () => {
      this.pauseListAudio();
    };
  },
  methods: {
    // 初始化音乐时长和监听播放进度
    setMusicDuration() {
      const musicEle = document.querySelector('#recommendAudio');
      musicEle.ondurationchange = () => {
        // 刷新总时长
        const tsDuration = parseInt(musicEle.duration, 10);
        const m = parseInt(tsDuration / 60, 10);
        const s = tsDuration % 60;
        this.musicDuration = `${m < 10 ? `0${m}` : m}:${s < 10 ? `0${s}` : s}`;
        console.log(`总时长ms：${tsDuration}; 分钟m: ${m}; 秒钟s：${s}`);
      };
      musicEle.ontimeupdate = () => {
        // 刷新进度条进度
        const pre = (musicEle.currentTime / musicEle.duration) * 100;
        if (pre <= 100) {
          $('.progress-ing').css('width', `${pre}%`);
        }
        // 刷新当前播放时间
        const tsCurTime = parseInt(musicEle.currentTime, 10);
        const m = parseInt(tsCurTime / 60, 10);
        const s = tsCurTime % 60;
        this.musicCurTime = `${m < 10 ? `0${m}` : m}:${s < 10 ? `0${s}` : s}`;
      };
      musicEle.onended = () => {
        this.pauseListAudio();
        musicEle.currentTime = 0;
      };
      $(document).on('touchend', () => {
        console.log('document addEventListen touchend');
        musicEle.play();
        musicEle.pause();
        $(document).off('touchend');
      });
    },
    // 播放音乐
    playListAudio(addlog) {
      $(document).off('touchend'); // 如果进入页面立刻点击播放需取消click事件监听
      if (addlog) log.log('list_play');
      const musicEle = document.querySelector('#recommendAudio');
      musicEle.play();
      this.isMusicPlaying = true;
      // 设定10s超时处理
      this.musicLoadTimer = setTimeout(() => {
      if (musicEle.duration > 1) {
        clearTimeout(this.musicLoadTimer);
      } else {
        console.log('亲的网络不太好哦, 请检查后重试~');
        this.pauseListAudio();
      }
      }, 10000);
    },
    // 暂停音乐
    pauseListAudio(addlog) {
      if (addlog) log.log('list_stop');
      const musicEle = document.querySelector('#recommendAudio');
      musicEle.pause();
      this.isMusicPlaying = false;
      // 清除10s超时处理
      if (this.musicLoadTimer) {
        clearTimeout(this.musicLoadTimer);
      }
    },
    // 点击进度条
    clickPrograssBar(e) {
      const pre = e.offsetX / $('.voice-progress').offset().width;
      if (pre <= 100) {
        const musicEle = document.querySelector('#recommendAudio');
        musicEle.currentTime = parseInt(musicEle.duration * pre, 10);
      }
    },
    // 拖拽音乐小圆点
    dragProgressPoint(e) {
      const barInfo = $('.voice-progress').offset();
      const moveDis = e.touches[0].clientX - barInfo.left;
      if (moveDis > 0 && moveDis < barInfo.width) {
        const pre = moveDis / barInfo.width;
        if (pre <= 100) {
          const musicEle = document.querySelector('#recommendAudio');
          musicEle.currentTime = parseInt(musicEle.duration * pre, 10);
        }
      }
    },
  }
</script>

# audio自动播放版

/*
 * 在部分手机上有bug
 */
<template>
  <div>
    <div class="swiper-container">
      <div class="swiper-wrapper">
        <div class="swiper-slide voice" v-for="(item, i) in cardList" :key="i">
          <p class="voice-name">{{item.author}}</p>
          <p class="voice-intro">{{item.short_desc}}</p>
          <div class="voice-avatar" :style="'background-image: url('+item.short_image+')'">
            <div class="voice-pause" @click.self="pauseAudio" v-if="isPalying&&activeIndex===i"></div>
            <div class="voice-play"  @click.self="playAudio"  v-else></div>
          </div>
          <div class="voice-listen" @click="showVoiceList(i)"></div>
        </div>
      </div>
      <div class="swiper-pagination">
        <div
          v-for="i in cardList.length"
          :key="i"
          :class="{'my-bullet': true, 'my-bullet-active': activeIndex === i-1}"
        ></div>
      </div>
      <audio id="swiperAudio" :src="musicSrc"></audio>
    </div>
  </div>
</template>

<script>
import Swiper from 'swiper4';
import $ from 'zepto';

export default {
  props: ['cardList'],
  data() {
    return {
      swiper: null,
      activeIndex: 0,
      limitAutoPlay: false,
      isPalying: false, // 控制音乐的播放、暂停图标切换
      isAutoPlay: true, // 控制轮播结束后是否自动播放
      canAutoPlayFlag: true,
      musicSrc: '',
      musicLoadTimer: null,
    };
  },
  mounted() {
    this.createSwiper();
    // 音乐自动播放
    const musicEle = document.querySelector('#swiperAudio');
    $(document).on('touchend', () => {
      console.log('index document addEventListen touchend');
      musicEle.play();
      musicEle.pause();
      $(document).off('touchend');
    });
    // 监听音乐结束
    musicEle.onended = () => {
      this.pauseAudio();
    };
    // pagehide暂停音乐
    window.onpagehide = () => {
      this.pauseAudio();
    };
  },
  methods: {
    // 轮播图滑动时切换音乐
    changeAudio() {
      // this.pauseAudio();
      this.clearTenTimerOut();
      this.musicSrc = this.cardList[this.activeIndex].short_voice;
      this.listenDuration();
    },
    // 监听音乐时长改变自动播放
    listenDuration() {
      if (this.canAutoPlayFlag) {
        const musicEle = document.querySelector('#swiperAudio');
        musicEle.ondurationchange = () => {
          if (this.isAutoPlay) {
            console.log(`duration：${musicEle.duration}  自动播放`);
            this.clearTenTimerOut();
            this.playAudio();
          }
        };
        this.canAutoPlayFlag = false;
      }
    },
    // 播放音乐
    playAudio() {
      if (this.limitAutoPlay) return;
      const musicEle = document.querySelector('#swiperAudio');
      if (musicEle) {
        musicEle.play();
        this.isPalying = true;
      }
      // 设定10s超时处理
      this.musicLoadTimer = setTimeout(() => {
        if (musicEle.duration > 1) {
          clearTimeout(this.musicLoadTimer);
        } else {
          console.log('亲的网络不太好哦, 请检查后重试~');
          this.pauseAudio();
        }
      }, 10000);
    },
    // 暂停音乐
    pauseAudio() {
      this.clearTenTimerOut();
      const musicEle = document.querySelector('#swiperAudio');
      if (musicEle) {
        musicEle.currentTime = 0;
        musicEle.pause();
        this.isPalying = false;
      }
    },
    // 清除10s超时处理
    clearTenTimerOut() {
      if (this.musicLoadTimer) {
        clearTimeout(this.musicLoadTimer);
      }
    },
    // 创建轮播图
    createSwiper() {
      const self = this;
      this.swiper = new Swiper('.swiper-container', {
        touchRatio: 0.6,
        resistanceRatio: 0,
        centeredSlides: true,
        effect: 'coverflow',
        coverflowEffect: {
          rotate: 0,
          stretch: -75,
          depth: 1200,
          modifier: 1,
          slideShadows: false,
        },
        on: {
          init() {
            self.musicSrc = self.cardList[this.activeIndex].short_voice;
          },
          slideChange() {
            self.activeIndex = this.activeIndex;
            self.$parent.recordActiveIndex(this.activeIndex);
            self.changeAudio();
          },
        },
      });
    },
    // 显示语音列表页
    showVoiceList(index) {
      this.limitAutoPlay = true;
      this.$parent.showVoiceList(this.cardList[index]);
    }
  }
};
</script>
